public final class GetBundleAction implements PrivilegedAction
{
  private String bundleName;
  private Locale locale;
  
  public GetBundleAction(String aBundleName, Locale aLocale)
  {
    bundleName = aBundleName;
    locale      = aLocale;
  }
  /* (non-Javadoc)
   * @see java.security.PrivilegedAction#run()
   */
  public Object run()
  {
    return ResourceBundle.getBundle(bundleName, locale);
  }

}
